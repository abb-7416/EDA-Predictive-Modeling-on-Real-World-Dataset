# Preprocessing pipelines
numeric_transformer = Pipeline(steps=[
    ('imputer', SimpleImputer(strategy='median')),
    ('scaler', StandardScaler())
])

categorical_transformer = Pipeline(steps=[
    ('imputer', SimpleImputer(strategy='most_frequent')),
    ('onehot', OneHotEncoder(handle_unknown='ignore', sparse_output=False))
])

preprocessor = ColumnTransformer(
    transformers=[
        ('num', numeric_transformer, numerical_feats),
        ('cat', categorical_transformer, categorical_feats)
    ])

# Feature importance from Random Forest (requires mapping back feature names)
# Get feature names after column transformer
def get_feature_names_from_column_transformer(column_transformer):
    # numeric features are passed through scaler -> keep their names
    num_features = numerical_feats
    # categorical features expanded by OneHotEncoder
    ohe = column_transformer.named_transformers_['cat'].named_steps['onehot']
    cat_ohe_names = list(ohe.get_feature_names_out(categorical_feats))
    return num_features + cat_ohe_names

feat_names = get_feature_names_from_column_transformer(preprocessor)
